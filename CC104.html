Algorithms (10)

1. Linear Search

for i in range(n):
    if arr[i] == target:
        return i
return -1

2. Binary Search

while low <= high:
    mid = (low + high) // 2
    if arr[mid] == target:
        return mid
    elif arr[mid] < target:
        low = mid + 1
    else:
        high = mid - 1
return -1

3. Bubble Sort

for i in range(n):
    for j in range(0, n-i-1):
        if arr[j] > arr[j+1]:
            arr[j], arr[j+1] = arr[j+1], arr[j]

4. Selection Sort

for i in range(n):
    min_idx = i
    for j in range(i+1, n):
        if arr[j] < arr[min_idx]:
            min_idx = j
    arr[i], arr[min_idx] = arr[min_idx], arr[i]

5. Insertion Sort

for i in range(1, n):
    key = arr[i]
    j = i - 1
    while j >= 0 and arr[j] > key:
        arr[j+1] = arr[j]
        j -= 1
    arr[j+1] = key

6. Merge Sort

def merge_sort(arr):
    if len(arr) > 1:
        mid = len(arr)//2
        L = arr[:mid]
        R = arr[mid:]
        merge_sort(L)
        merge_sort(R)
        i = j = k = 0
        while i < len(L) and j < len(R):
            if L[i] < R[j]:
                arr[k] = L[i]
                i += 1
            else:
                arr[k] = R[j]
                j += 1
            k += 1
        while i < len(L):
            arr[k] = L[i]
            i += 1
            k += 1
        while j < len(R):
            arr[k] = R[j]
            j += 1
            k += 1

7. Quick Sort

def quicksort(arr):
    if len(arr) <= 1:
        return arr
    pivot = arr[len(arr)//2]
    left = [x for x in arr if x < pivot]
    mid = [x for x in arr if x == pivot]
    right = [x for x in arr if x > pivot]
    return quicksort(left) + mid + quicksort(right)

8. Factorial

def factorial(n):
    result = 1
    for i in range(1, n+1):
        result *= i
    return result

9. GCD (Euclid Algorithm)

def gcd(a, b):
    while b != 0:
        a, b = b, a % b
    return a

Array Implementation

Definition

An array is a fixed-size, indexed collection of elements stored in contiguous memory.

Examples

arr = [10, 20, 30, 40]
print(arr[1])  # 20
print(len(arr))  # 4

Stacks (Infix, Prefix, Postfix)

Definition

A stack is a LIFO structure used in expression conversion and function calls.

Python Example

stack = []
stack.append(5)
stack.append(9)
print(stack.pop())  # 9

Expression Example

Infix:   A + B * C
Prefix:  +A*BC
Postfix: ABC*+

Queue Implementation

Definition

A queue is a FIFO structure used in scheduling, buffering, and BFS.

Example

from collections import deque
q = deque([1, 2, 3])
q.append(4)
print(q.popleft())  # 1

Binary Tree Traversals

Definition

A tree is a hierarchical structure. Traversals visit nodes in a specific order.

Traversal Types

Inorder:   Left → Root → Right
Preorder:  Root → Left → Right
Postorder: Left → Right → Root

Python Example

def preorder(node):
    if node:
        print(node.val)
        preorder(node.left)
        preorder(node.right)
